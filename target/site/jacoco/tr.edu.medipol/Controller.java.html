package tr.edu.medipol;

import org.springframework.web.bind.annotation.*;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

@RestController
@RequestMapping("/employees")
class Controller {

    public record Employee(String number, String name) {}

    public record Task(String employeeNumber, String task) {}

    public static List<Employee> Employees = new ArrayList<>();
    public static Map<String, List<String>> Tasks = new HashMap<>();

    static {
        Employees.add(new Employee("1", "Ali"));
        Employees.add(new Employee("2", "Berk"));
        Employees.add(new Employee("3", "Özkan"));

        Tasks.put("1", new ArrayList<>());
        Tasks.put("2", new ArrayList<>());
        Tasks.put("3", new ArrayList<>());
    }

    @GetMapping("/listTasks")
    public Map<String, List<String>> listAllTasks() {
        return Tasks;
    }

    @PostMapping("/addTask")
    public String addTask(@RequestBody Task eTask) {
        String employeeNumber = eTask.employeeNumber();
        String task = eTask.task();

        List<String> tasks = Tasks.get(employeeNumber);
        if (tasks == null)
            return "Çalışan bulunamadı";

        tasks.add(task);
        return "Görev eklendi";
    }
}
